import cv2 import numpy as np import mediapipe as mp mp_hands = mp.solutions.hands hands = mp_hands.Hands() mp_draw = mp.solutions.drawing_utils cap = cv2.VideoCapture(0) canvas = np.zeros((480, 640, 3), dtype=np.uint8) prev_x, prev_y = 0, 0 def fingers_up(handLms): fingers = [] # Index if handLms.landmark[8].y < handLms.landmark[6].y: fingers.append(1) else: fingers.append(0) # Middle if handLms.landmark[12].y < handLms.landmark[10].y: fingers.append(1) else: fingers.append(0) # Ring if handLms.landmark[16].y < handLms.landmark[14].y: fingers.append(1) else: fingers.append(0) # Pinky if handLms.landmark[20].y < handLms.landmark[18].y: fingers.append(1) else: fingers.append(0) return fingers while True: ret, frame = cap.read() frame = cv2.flip(frame, 1) rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB) result = hands.process(rgb) if result.multi_hand_landmarks: for handLms in result.multi_hand_landmarks: h, w, c = frame.shape x = int(handLms.landmark[8].x * w) y = int(handLms.landmark[8].y * h) finger_state = fingers_up(handLms) index_up = finger_state[0] middle_up = finger_state[1] ring_up = finger_state[2] pinky_up = finger_state[3] # ✏️ DRAW MODE (only index finger) if index_up and not middle_up and not ring_up and not pinky_up: cv2.circle(frame, (x, y), 8, (0, 255, 0), -1) if prev_x == 0 and prev_y == 0: prev_x, prev_y = x, y cv2.line(canvas, (prev_x, prev_y), (x, y), (255, 255, 255), 5) prev_x, prev_y = x, y # ✋ STOP MODE (index + middle) elif index_up and middle_up and not ring_up and not pinky_up: cv2.circle(frame, (x, y), 10, (255, 0, 0), -1) prev_x, prev_y = 0, 0 # 🧽 ERASE MODE (all fingers open) elif index_up and middle_up and ring_up and pinky_up: cv2.circle(frame, (x, y), 20, (0, 0, 255), -1) cv2.circle(canvas, (x, y), 25, (0, 0, 0), -1) prev_x, prev_y = 0, 0 else: prev_x, prev_y = 0, 0 mp_draw.draw_landmarks(frame, handLms, mp_hands.HAND_CONNECTIONS) combined = cv2.add(frame, canvas) cv2.imshow("Gesture Drawing Board", combined) if cv2.waitKey(1) & 0xFF == 27: break cap.release() cv2.destroyAllWindows()
