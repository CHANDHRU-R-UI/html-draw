import cv2 import numpy as np import mediapipe as mp mp_hands = mp.solutions.hands hands = mp_hands.Hands( max_num_hands=1, min_detection_confidence=0.7, min_tracking_confidence=0.7 ) mp_draw = mp.solutions.drawing_utils cap = cv2.VideoCapture(0) canvas = np.zeros((480, 640, 3), dtype=np.uint8) prev_x, prev_y = 0, 0 def fingers_up(handLms): fingers = [] //Index fingers.append(1 if handLms.landmark[8].y < handLms.landmark[6].y else 0) //Middle fingers.append(1 if handLms.landmark[12].y < handLms.landmark[10].y else 0) //Ring fingers.append(1 if handLms.landmark[16].y < handLms.landmark[14].y else 0) //Pinky fingers.append(1 if handLms.landmark[20].y < handLms.landmark[18].y else 0) return fingers while True: ret, frame = cap.read() if not ret: break frame = cv2.flip(frame, 1) h, w, c = frame.shape rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB) result = hands.process(rgb) if result.multi_hand_landmarks: for handLms in result.multi_hand_landmarks: x = int(handLms.landmark[8].x * w) y = int(handLms.landmark[8].y * h) finger_state = fingers_up(handLms) index_up = finger_state[0] middle_up = finger_state[1] ring_up = finger_state[2] pinky_up = finger_state[3] // DRAW MODE (Only Index Finger) if index_up and not middle_up and not ring_up and not pinky_up: cv2.circle(frame, (x, y), 8, (0, 255, 0), -1) # Start new stroke properly if prev_x == 0 and prev_y == 0: prev_x, prev_y = x, y # Distance check to avoid jump lines distance = np.hypot(x - prev_x, y - prev_y) if distance < 60: cv2.line(canvas, (prev_x, prev_y), (x, y), (255, 255, 255), 7) prev_x, prev_y = x, y //ERASE MODE (Index + Middle) elif index_up and middle_up and not ring_up and not pinky_up: cv2.circle(frame, (x, y), 20, (0, 0, 255), -1) cv2.circle(canvas, (x, y), 30, (0, 0, 0), -1) prev_x, prev_y = 0, 0 // STOP MODE (Any Other Gesture) else: prev_x, prev_y = 0, 0 mp_draw.draw_landmarks(frame, handLms, mp_hands.HAND_CONNECTIONS) //Merge canvas and frame combined = cv2.addWeighted(frame, 0.8, canvas, 1, 0) cv2.imshow("Gesture Drawing Board - Fast Mode", combined) if cv2.waitKey(1) & 0xFF == 27: break cap.release() cv2.destroyAllWindows()
